    /* user-utils.asm */
    /* int writeTo(int fd, void* buffer, int size, int offset) */
    ENTRY(writeTo)
        pushl %ebp
        movl %esp, %ebp
        pushl %ebx;  // Save EBX, ESI and EDI if modified
        movl $40, %eax // syscall number
        movl 0x8(%ebp), %ebx;	//fd
        movl 0xC(%ebp), %ecx;	//buffer
        movl 0x10(%ebp), %edx;	//size
        movl 0x14(%ebp), %esi;	//offset
        call syscall_sysenter
        popl %ebx
        test %eax, %eax
        popl %ebp
        ret

    /* int readFrom(int fd, void* buffer, int size, int offset) */
    ENTRY(readFrom)
        pushl %ebp
        movl %esp, %ebp
        pushl %ebx;  // Save EBX, ESI and EDI if modified
        movl $41, %eax // syscall number
        movl 0x8(%ebp), %ebx;	//fd
        movl 0xC(%ebp), %ecx;	//buffer
        movl 0x10(%ebp), %edx;	//size
        movl 0x14(%ebp), %esi;	//offset
        call syscall_sysenter
        popl %ebx
        test %eax, %eax
        popl %ebp
        ret

    - sched.h : Añadida la estructura de communication_struct dentro de task_struct
    - user-utils.S : Implementación de los wrappers a writeTo y readFrom, 
        usando la instrucción ENTRY(syscall_name) 
    - sys_call_table.S : Definición de los números de syscall para writeTo(40) y readFrom(41) 
    - sys.c : Inicializacion de las nuevas syscalls, posteriormente seran implementadas en sys_writeTo y sys_readFrom

    //mm.h 
    void mapPhysicalPageFrom(struct task_struct *t, unsigned int address);

    //mm.c 
    void mapPhysicalPageFrom(struct task_struct *t, unsigned int address)
    {
        //Obtener TP del proceso, marco fisico 
        page_table_entry * process_PT =  get_PT(t);

        //Obtener TP del proceso actual
        page_table_entry * current_PT = get_PT(current());

        //Obtener el frame de la pagina logica COMM_PAGE_LOG del proceso t
        unsigned int frame = get_frame(process_PT, COMM_PAGE_ID);
        
        //Mapear la pagina fisica en la direccion logica COMM_PAGE_LOG del proceso actual
        set_ss_pag(current_PT, COMM_PAGE_ID, frame);
        
        //Flush tlb
        set_cr3(get_DIR(current()));
    }
